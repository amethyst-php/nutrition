Create

Define a new instance of the Manager

use Railken\Amethyst\Managers\BarManager;

$manager = new BarManager();

Create a new entity

$params = [
    "name" => "Mr. Arlo Monahan MD",
    "description" => "Ut tempore voluptatem omnis soluta. Aspernatur ab illum laboriosam enim dolorem est. Veritatis dignissimos exercitationem consequatur explicabo illum quidem repellat."
];

$result = $manager->create($params);

Check the result of the operation

if ($result->ok()) {
    // All ok
} else {
    // Something goes wrong
}

Retrieve the entity from the result

$entity = $result->getResource();

Throw an exception immediately if the operation fails

use Railken\Lem\Exceptions\Exception;

$params = [
    "name" => "Mr. Arlo Monahan MD",
    "description" => "Ut tempore voluptatem omnis soluta. Aspernatur ab illum laboriosam enim dolorem est. Veritatis dignissimos exercitationem consequatur explicabo illum quidem repellat."
];

try {
    $result = $manager->createOrFail($params);
} catch (Exception $e) {
    // ...
}
