Create

Define a new instance of the Manager

use Railken\Amethyst\Managers\FooManager;

$manager = new FooManager();

Create a new entity

$params = [
    "name" => "Elva Marvin",
    "description" => "Totam rerum rem et porro tenetur maiores enim. Et rerum totam veritatis. Error dolor numquam odit repudiandae. Debitis eum tenetur totam libero.",
    "bar" => [
        "name" => "Dax Ruecker",
        "description" => "Aliquam et officia dignissimos ut. Unde voluptatem illum vero doloremque. Nihil veniam quia tenetur adipisci. Ut sunt enim consequatur et ipsam."
    ]
];

$result = $manager->create($params);

Check the result of the operation

if ($result->ok()) {
    // All ok
} else {
    // Something goes wrong
}

Retrieve the entity from the result

$entity = $result->getResource();

Throw an exception immediately if the operation fails

use Railken\Lem\Exceptions\Exception;

$params = [
    "name" => "Elva Marvin",
    "description" => "Totam rerum rem et porro tenetur maiores enim. Et rerum totam veritatis. Error dolor numquam odit repudiandae. Debitis eum tenetur totam libero.",
    "bar" => [
        "name" => "Dax Ruecker",
        "description" => "Aliquam et officia dignissimos ut. Unde voluptatem illum vero doloremque. Nihil veniam quia tenetur adipisci. Ut sunt enim consequatur et ipsam."
    ]
];

try {
    $result = $manager->createOrFail($params);
} catch (Exception $e) {
    // ...
}
