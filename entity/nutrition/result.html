Result

Once you've got the result you should always check if an error has occurred, if not, retrieve the resource.

use Railken\Amethyst\Managers\NutritionManager;

$manager = new NutritionManager();

$result = $manager->create([
    "calories" => 20.3,
    "salt" => 20.3,
    "protein" => 20.3,
    "fibre" => 20.3,
    "carbohydrate" => 20.3,
    "sugars" => 20.3,
    "fat" => 20.3,
    "saturates" => 20.3,
    "nutritionable_type" => "foo",
    "nutritionable" => [
        "name" => "Lexus DuBuque",
        "description" => "Ut mollitia cumque sed natus qui dolor eaque. Nihil hic sunt voluptas eum et nam. Eum sunt est illo omnis.",
        "bar" => [
            "name" => "Dr. Kaylie Heaney V",
            "description" => "Ut incidunt asperiores ullam. Rerum sequi enim aliquam illo. Facere repellendus ex maiores."
        ]
    ]
]);

if ($result->ok()) {

    $resource = $result->getResource();

} else {

    // Loop through all errors
    $result->getErrors()->map(function($error) {
        return $error->toArray();
    }))

    // Retrieve an array of all errors
    $result->getSimpleErrors();

    /* The result is something like this:

        [0] => Array
            (
                [code] => FIELD_NOT_DEFINED
                [label] => field
                [message] => The field is required
                [value] =>
            )
    */

}
